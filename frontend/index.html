const express = require("express"); const cors = require("cors"); require("dotenv").config(); const app = express(); // âœ… Allow requests from anywhere (safe for MVP) app.use(cors()); // Parse JSON bodies app.use(express.json()); // Health check route app.get("/", (req, res) => { res.json({ status: "ok", service: "justcheck-backend" }); }); // Render provides PORT const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log(`Server running on port ${PORT}`); });
